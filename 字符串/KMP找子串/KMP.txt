#include <iostream>
#include <cstring>
using namespace std;
#define  N 100
char P[N];
char T[N];
int next[N];
void Next(int len)
{
	int i,j;
	next[0]=-1;
	for (i=1;i<len;i++)
	{
		j=next[i-1];
		while(j>=0&&P[j+1]!=P[i])
			j=next[j];
		next[i]=(P[j+1]==P[i])?j+1:-1;
	}
}
